Want to Auto Cookie Clicker?

This is a JavaScript file that will auto-click in Cookie Clicker. Here's how to paste the code in:

  1. Go to Cookie Clicker.
  2. Double click anywhere on the screen.
  3. Click "Inspect" and go to the Console tab.
  4. Paste the following code in:
(function autoClickBuyAndAscend() {
    // Change the player's name to "ur welcome blud"
    Game.bakeryNameSet("ur welcome blud");

    // Auto-click on the big cookie
    setInterval(function() {
        document.getElementById('bigCookie').click();
    }, 10);  // Adjust the interval (10 milliseconds) for clicking the cookie

    // Auto-buy store upgrades
    setInterval(function() {
        for (var i = 0; i < Game.UpgradesInStore.length; i++) {
            var upgrade = Game.UpgradesInStore[i];
            if (Game.cookies >= upgrade.getPrice()) {
                upgrade.buy();  // Buy the upgrade if you can afford it
                console.log("Bought upgrade:", upgrade.name);
                return;  // Exit the loop once an upgrade is bought
            }
        }
    }, 1000);  // Check every 1 second for available store upgrades

    // Auto-buy buildings (cursors, grandmas, etc.)
    setInterval(function() {
        for (var j = 0; j < Game.ObjectsById.length; j++) {
            var building = Game.ObjectsById[j];
            if (Game.cookies >= building.price) {
                building.buy();  // Buy the building if you can afford it
                console.log("Bought building:", building.name);
                return;  // Exit the loop once a building is bought
            }
        }
    }, 1000);  // Check every 1 second for available buildings to buy

    // Check if cookies have reached 85.184 quintillion (8.5184e+19)
    setInterval(function() {
        var targetCookies = 8.5184e+19;  // 85.184 quintillion cookies
        if (Game.cookies >= targetCookies) {
            console.log("Target cookies reached! Clicking Legacy to ascend...");

            // Check if the Legacy button is visible (usually has the id 'legacyButton')
            if (document.getElementById('legacyButton') && document.getElementById('legacyButton').style.display != 'none') {
                document.getElementById('legacyButton').click();  // Click the Legacy button
                console.log("Ascended!");
            }
        }
    }, 1000);  // Check every 1 second for cookie count and ascension
})();
